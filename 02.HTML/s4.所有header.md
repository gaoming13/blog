## 所有的header部分

```html
<!-- 标题 -->
<title>网页标题</title>
<!-- 页面编码 -->
<meta charset="utf-8">
<!-- 视口大小 -->
<meta name="viewport" content="width=device-width, initial-scale=1">
<!-- 禁止谷歌翻译 -->
<meta name="google" content="notranslate">
<!-- 渲染内核 -->
<meta name="renderer" content="webkit">
<!-- 描述 -->
<meta name="description" content="这是描述的内容">
<!-- 关键词 -->
<meta name="keywords" content="这是关键词">

<meta property="og:title" content="这是标题" />
<meta property="og:url" content="这是链接" />
<meta property="og:description" content="这是描述" />
<meta property="og:locale" content="en" />
<meta property="og:locale:alternate" content="en" />
<meta property="og:locale:alternate" content="ja" />
<meta property="og:locale:alternate" content="zh-Hans" />

<!-- icon -->
<link rel="shortcut icon" href="favicon.ico" />
<link rel="shortcut icon" sizes="512x512" href="logo-512.png" />
<link rel="apple-touch-icon" href="logo-512.png" />
<!-- css -->
<link href="app.css" rel="stylesheet">
```
